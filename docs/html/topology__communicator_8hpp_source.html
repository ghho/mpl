307 void neighbor_alltoallv(const T *senddata, const layouts<T> &sendls,
308 const displacements &senddispls, T *recvdata,
309 const layouts<T> &recvls, const displacements &recvdispls) const {
310 const std::vector<int> counts(recvls.size(), 1);
311 static_assert(
312 sizeof(decltype(*sendls())) == sizeof(MPI_Datatype),
313 "compiler adds some unexpected padding, reinterpret cast will yield wrong results");
314 MPI_Neighbor_alltoallw(senddata, counts.data(), senddispls(),
315 reinterpret_cast<const MPI_Datatype *>(sendls()), recvdata,
316 counts.data(), recvdispls(),
317 reinterpret_cast<const MPI_Datatype *>(recvls()), comm_);
318 }
