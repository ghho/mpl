129 explicit graph_communicator(const communicator &other, const edge_set &edges,
130 bool reorder = true) {
131 int nodes{0};
132 for (const auto &e : edges) {
133#if defined MPL_DEBUG
134 if (std::get<0>(e) < 0 or std::get<1>(e) < 0)
135 throw invalid_argument();
136#endif
137 nodes = std::max({nodes, std::get<0>(e) + 1, std::get<1>(e) + 1});
138 }
139 std::vector<int> edges_list, index(nodes, 0);
140 edges_list.reserve(edges.size());
141 // the following works because the edge set is ordered with respect to the pairs of
142 // edge-node numbers
143 for (const auto &e : edges) {
144 edges_list.push_back(std::get<1>(e));
145 ++index[std::get<0>(e)];
146 }
147 std::partial_sum(index.begin(), index.end(), index.begin());
148 MPI_Graph_create(other.comm_, nodes, index.data(), edges_list.data(), reorder, &comm_);
149 }

158 graph_communicator &operator=(const graph_communicator &other) noexcept {
159 if (this != &other) {
160 if (is_valid()) {
161 int result_1;
162 MPI_Comm_compare(comm_, MPI_COMM_WORLD, &result_1);
163 int result_2;
164 MPI_Comm_compare(comm_, MPI_COMM_SELF, &result_2);
165 if (result_1 != MPI_IDENT and result_2 != MPI_IDENT)
166 MPI_Comm_free(&comm_);
167 }
168 MPI_Comm_dup(other.comm_, &comm_);
169 }
170 return *this;
