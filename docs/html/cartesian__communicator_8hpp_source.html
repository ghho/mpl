152 class dimensions {

154
155 public:
156 class dimension_periodicity_proxy {
157 int &dim_;
158 int &is_periodic_;
159
160 dimension_periodicity_proxy(int &dim, int &is_periodic)
161 : dim_{dim}, is_periodic_{is_periodic} {}
162
163 public:
164 template<std::size_t N>
165 [[nodiscard]] decltype(auto) get() const {
166 if constexpr (N == 0)
167 return dim_ * 1;
168 else if constexpr (N == 1)
169 return is_periodic_ == 0 ? periodicity_tag::non_periodic
170 : periodicity_tag::periodic;
171 }
172
173 dimension_periodicity_proxy &operator=(const std::tuple<int, periodicity_tag> &t) {
174 dim_ = std::get<int>(t);
175 is_periodic_ = std::get<periodicity_tag>(t) == periodic;
176 return *this;
177 }
178
179 bool operator==(const std::tuple<int, periodicity_tag> &t) const {
180 return dim_ == std::get<int>(t) and
181 (static_cast<bool>(is_periodic_) ==
182 (std::get<periodicity_tag>(t) == periodicity_tag::periodic));
183 }
184
185 bool operator!=(const std::tuple<int, periodicity_tag> &t) const {
186 return not(*this == t);
187 }
188
189 friend dimensions;
190 };

398 explicit cartesian_communicator(const cartesian_communicator &other,
399 const included_tags &is_included) {
400#if defined MPL_DEBUG
401 if (is_included.size() != other.dimensionality())
402 throw invalid_size();
403#endif
404 MPI_Cart_sub(other.comm_, reinterpret_cast<const int *>(is_included.data()), &comm_);

409 cartesian_communicator(cartesian_communicator &&other) noexcept {
410 comm_ = other.comm_;
411 other.comm_ = MPI_COMM_SELF;
412 }
413
421 cartesian_communicator &operator=(const cartesian_communicator &other) noexcept {
422 if (this != &other) {
423 if (is_valid()) {
424 int result_1;
425 MPI_Comm_compare(comm_, MPI_COMM_WORLD, &result_1);
426 int result_2;
427 MPI_Comm_compare(comm_, MPI_COMM_SELF, &result_2);
428 if (result_1 != MPI_IDENT and result_2 != MPI_IDENT)
429 MPI_Comm_free(&comm_);
430 }
431 MPI_Comm_dup(other.comm_, &comm_);
432 }
433 return *this;
434 }
435
440 cartesian_communicator &operator=(cartesian_communicator &&other) noexcept {
441 if (this != &other) {
442 int result_1{0}, result_2{0};
443 MPI_Comm_compare(comm_, MPI_COMM_WORLD, &result_1);
444 MPI_Comm_compare(comm_, MPI_COMM_SELF, &result_2);
445 if (result_1 != MPI_IDENT and result_2 != MPI_IDENT)
446 MPI_Comm_free(&comm_);
447 comm_ = other.comm_;
448 other.comm_ = MPI_COMM_SELF;
449 }
450 return *this;
451 }
