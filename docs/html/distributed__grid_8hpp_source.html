163 explicit distributed_grid(const cartesian_communicator &communicator,
164 const dimensions &dims) {
165#if defined MPL_DEBUG
166 if (communicator.dimensionality() != dim or global_size_.size() != dim or
167 overlap_.size() != dim)
168 throw invalid_dim();
169#endif
170 const auto c_dimensions{communicator.get_dimensions()};
171 const auto c_coord{communicator.coordinates()};
172 size_type volume{1};
173 for (std::size_t i{0}; i < dim; ++i) {
174 global_size_[i] = dims[i].size;
175 overlap_[i] = dims[i].overlap;
176 global_begin_[i] = global_begin(global_size_[i], c_dimensions.size(i), c_coord[i]);
177 global_end_[i] = global_end(global_size_[i], c_dimensions.size(i), c_coord[i]);
178 size_[i] = global_end_[i] - global_begin_[i];
179 overlap_end_[i] = size_[i] + 2 * overlap_[i];
180 volume *= size_[i] + 2 * overlap_[i];
181 }
182 v_.resize(volume);
183 left_mirror_layout_.reserve(dim);
184 right_mirror_layout_.reserve(dim);
185 left_border_layout_.reserve(dim);
186 right_border_layout_.reserve(dim);
187 for (std::size_t i{0}; i < dim; ++i) {
188 typename subarray_layout<T>::parameter par_m_l, par_m_r, par_b_l, par_b_r;
189 for (std::size_t j{dim - 1}; true; --j) {
190 if (j == i) {
191 par_m_l.add(overlap_end_[j], overlap_[j], 0);
192 par_m_r.add(overlap_end_[j], overlap_[j], size_[j] + overlap_[j]);
193 par_b_l.add(overlap_end_[j], overlap_[j], overlap_[j]);
194 par_b_r.add(overlap_end_[j], overlap_[j], size_[j]);
195 } else {
196 par_m_l.add(overlap_end_[j], size_[j], overlap_[j]);
197 par_m_r.add(overlap_end_[j], size_[j], overlap_[j]);
198 par_b_l.add(overlap_end_[j], size_[j], overlap_[j]);
199 par_b_r.add(overlap_end_[j], size_[j], overlap_[j]);
200 }
201 if (j == 0)
202 break;
203 }
204 left_mirror_layout_.push_back(subarray_layout<T>(par_m_l));
205 right_mirror_layout_.push_back(subarray_layout<T>(par_m_r));
206 left_border_layout_.push_back(subarray_layout<T>(par_b_l));
207 right_border_layout_.push_back(subarray_layout<T>(par_b_r));
208 typename subarray_layout<T>::parameter par_i;
209 for (std::size_t j{dim - 1}; true; --j) {
210 par_i.add(overlap_end_[j], size_[j], overlap_[j]);
211 if (j == 0)
212 break;
213 }
214 interior_layout_ = subarray_layout<T>(par_i);
215 }
216 }

546 local_grid(const cartesian_communicator &communicator, const dimensions &dims)
547 : global_size_(dims.size_) {
548#if defined MPL_DEBUG
549 if (communicator.dimensionality() != dim or global_size_.size() != dim)
550 throw invalid_dim();
551#endif
552 size_type volume{1};
553 for (std::size_t i{0}; i < dim; ++i)
554 volume *= global_size_[i];
555 v_.resize(volume);
556 const auto c_dimensions{communicator.get_dimensions()};
557 const auto c_size{communicator.size()};
558 for (int i{0}, i_end{c_size}; i < i_end; ++i) {
559 const auto coords{communicator.coordinates(i)};
560 typename subarray_layout<T>::parameter par;
561 for (std::size_t j{dim - 1}; true; --j) {
562 const auto begin{global_begin(global_size_[j], c_dimensions.size(j), coords[j])};
563 const auto end{global_end(global_size_[j], c_dimensions.size(j), coords[j])};
564 par.add(global_size_[j], end - begin, begin);
565 if (j == 0)
566 break;
567 }
568 sub_layouts_.push_back(subarray_layout<T>(par));
569 }
570 }
