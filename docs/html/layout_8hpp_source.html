81 private:
82 MPI_Datatype type_;
83
84 protected:
85 explicit layout(MPI_Datatype new_type) : type_(new_type) {
86 if (type_ != MPI_DATATYPE_NULL)
87 MPI_Type_commit(&type_);
88 }
89
90 public:
92 layout() : type_(MPI_DATATYPE_NULL) {}
93
97 layout(const layout &l) : type_(MPI_DATATYPE_NULL) {
98 if (l.type_ != MPI_DATATYPE_NULL)
99 MPI_Type_dup(l.type_, &type_);
100 }
101
105 layout(layout &&l) noexcept : type_(l.type_) { l.type_ = MPI_DATATYPE_NULL; }
106
110 layout &operator=(const layout &l) {
111 if (this != &l) {
112 if (type_ != MPI_DATATYPE_NULL)
113 MPI_Type_free(&type_);
114 if (l.type_ != MPI_DATATYPE_NULL)
115 MPI_Type_dup(l.type_, &type_);
116 else
117 type_ = MPI_DATATYPE_NULL;
118 }
119 return *this;
120 }
121

126 if (this != &l) {
127 if (type_ != MPI_DATATYPE_NULL)
128 MPI_Type_free(&type_);
129 type_ = l.type_;
130 l.type_ = MPI_DATATYPE_NULL;
131 }
132 return *this;
133 }
134
141 [[nodiscard]] MPI_Datatype native_handle() const { return type_; }
142
148 [[nodiscard]] ssize_t byte_extent() const {
149 MPI_Count lb_, extent_;
150 MPI_Type_get_extent_x(type_, &lb_, &extent_);
151 if (lb_ == MPI_UNDEFINED or extent_ == MPI_UNDEFINED)
152 throw invalid_datatype_bound();
153 return extent_;
154 }
155
162 [[nodiscard]] ssize_t byte_lower_bound() const {
163 MPI_Count lb_, extent_;
164 MPI_Type_get_extent_x(type_, &lb_, &extent_);

466 template<typename T>
467 class contiguous_layout : public layout<T> {
468 using layout<T>::type_;
469
470 static MPI_Datatype build(
471 size_t count, MPI_Datatype old_type = detail::datatype_traits<T>::get_datatype()) {
472 MPI_Datatype new_type;
473 if (count <= static_cast<size_t>(std::numeric_limits<int>::max())) {
474 MPI_Type_contiguous(static_cast<int>(count), old_type, &new_type);
475 } else {
476 const size_t modulus{std::numeric_limits<int>::max()};
477 const size_t count_1{count / modulus};
478 const size_t count_0{count - count_1 * modulus};
479 MPI_Count lb, extent;
480 MPI_Type_get_extent_x(old_type, &lb, &extent);
481 MPI_Datatype type_modulus;
482 MPI_Type_contiguous(static_cast<int>(modulus), old_type, &type_modulus);
483 std::vector<int> block_lengths{static_cast<int>(count_0), static_cast<int>(count_1)};
484#if defined MPL_DEBUG
485 if (count_0 * extent > static_cast<size_t>(std::numeric_limits<MPI_Aint>::max()))
486 throw invalid_size();
487#endif
488 std::vector<MPI_Aint> displacements{0, static_cast<MPI_Aint>(count_0 * extent)};
489 std::vector<MPI_Datatype> types{old_type, type_modulus};
490 MPI_Type_create_struct(2, block_lengths.data(), displacements.data(), types.data(),
491 &new_type);
492 }
493 return new_type;
494 }
495
496 size_t count_;
497
498 [[nodiscard]] size_t size() const { return count_; }
499
500 public:
503 explicit contiguous_layout(size_t count = 0) : layout<T>(build(count)), count_(count) {}
504
509 explicit contiguous_layout(size_t count, const contiguous_layout<T> &l)
510 : layout<T>(build(count, l.type_)), count_(l.count_ * count) {}
511
514 contiguous_layout(const contiguous_layout<T> &l) : layout<T>(l), count_(l.count_) {}
515
518 contiguous_layout(contiguous_layout &&l) noexcept
519 : layout<T>(std::move(l)), count_(l.count_) {
520 l.count_ = 0;
521 }
522
526 contiguous_layout<T> &operator=(const contiguous_layout<T> &l) {
527 if (&l != this) {
528 layout<T>::operator=(l);
529 count_ = l.count_;
530 }
531 return *this;
532 }
533
537 contiguous_layout<T> &operator=(contiguous_layout<T> &&l) noexcept {
538 if (this != &l) {
539 layout<T>::operator=(std::move(l));
540 count_ = l.count_;
541 l.count_ = 0;
542 }
543 return *this;
544 }
545
548 void swap(contiguous_layout<T> &other) {
549 std::swap(type_, other.type_);
550 std::swap(count_, other.count_);
551 }
552
553 using layout<T>::byte_extent;
554 using layout<T>::byte_lower_bound;
555 using layout<T>::byte_upper_bound;
556 using layout<T>::extent;
557 using layout<T>::lower_bound;
558 using layout<T>::upper_bound;
559
560 friend class impl::base_communicator;
561 friend class communicator;
562 friend class communicator;
563 friend class contiguous_layouts<T>;
564 };
