111 explicit distributed_graph_communicator(const communicator &other,
112 const neighbours_set &sources,
113 const neighbours_set &destinations,
114 bool reorder = true) {
115 std::vector<int> sources_vector;
116 std::vector<int> source_weights;
117 sources_vector.reserve(sources.size());
118 source_weights.reserve(sources.size());
119 for (const auto &x : sources) {
120 sources_vector.push_back(x.rank);
121 source_weights.push_back(x.weight);
122 }
123 std::vector<int> destinations_vector;
124 std::vector<int> destination_weights;
125 destinations_vector.reserve(destinations.size());
126 destination_weights.reserve(destinations.size());
127 for (const auto &x : destinations) {
128 destinations_vector.push_back(x.rank);
129 destination_weights.push_back(x.weight);
130 }
131 MPI_Dist_graph_create_adjacent(other.comm_, sources_vector.size(), sources_vector.data(),
132 source_weights.data(), destinations.size(),
133 destinations_vector.data(), destination_weights.data(),
134 MPI_INFO_NULL, reorder, &comm_);
135 }
